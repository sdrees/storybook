import type { Meta, StoryObj } from '@storybook/vue3';

import Layout from './Layout.vue';

  component: MyComponent,

  render: () => ({
    components: { Layout, MyComponent },
    template: `
      <Layout>
        <header>
          <h1>Example</h1>
        </header>
        <article>
          <MyComponent />
        </article>
      </Layout>
    `,
  }),
};
