import Table from './Table.vue';

  name="Numeric"
  args={{
    data: [
      [1, 2, 3],
      [4, 5, 6],
    ],
    size: 'large',
  }}
  render={(args) => ({
    components: { Table },
    setup() {
      return { args };
    },
    template: `
    <Table v-bind="args">
      <tr v-for="(row, idx1) in data">
        <td v-for="(col, idx2) in row">
          {{ data[idx1][idx2] }}
        </td>
      </tr>
    </Table>`,
  })} />
