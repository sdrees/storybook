import type { Meta, StoryObj } from '@storybook/svelte';

import Button from './Button.svelte';

};

 * See https://storybook.js.org/docs/svelte/api/csf

 */

  render: () => ({
    Component: Button,
    props: {
      backgroundColor: '#ff0',
      label: 'Button',
    },
  }),
};
