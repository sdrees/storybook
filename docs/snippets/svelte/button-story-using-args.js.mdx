import Button from './Button.svelte';

export default {

};

  render: (args) => ({
    Component: Button,
    props: args,

    label: 'Button',

};

  render: (args) => ({
    Component: Button,
    props: args,

    label: '😄👍😍💯',
  },
};

export const Tertiary = {
  render: (args) => ({
    Component: Button,
