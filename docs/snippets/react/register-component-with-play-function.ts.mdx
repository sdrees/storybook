import React from 'react';

export default {
  /* ðŸ‘‡ The title prop is optional.

    });

    const passwordInput = screen.getByLabelText('password', {
      selector: 'input',
    });

    await userEvent.type(passwordInput, 'ExamplePassword', {
      delay: 100,
    });

    const submitButton = screen.getByRole('button');

  },
};
