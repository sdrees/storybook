import { Meta, Story } from '@storybook/addon-docs';

import { Table } from './Table';

  <Table {...args} >

  </Table>
);

<Story
  name="Numeric"
  args={{
    data: [[1, 2, 3], [4, 5, 6]],
    size: 'large',
  }}>
