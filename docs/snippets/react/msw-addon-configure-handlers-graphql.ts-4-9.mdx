import { ApolloClient, ApolloProvider, InMemoryCache } from '@apollo/client';

      fetchPolicy: 'no-cache',
      errorPolicy: 'all',
    },
    query: {
      fetchPolicy: 'no-cache',
      errorPolicy: 'all',
    },
