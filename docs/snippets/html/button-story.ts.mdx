import { Meta, StoryFn } from '@storybook/html';

  title: 'Button',

export const Primary: StoryFn = (): HTMLButtonElement => {
  const btn = document.createElement('button');
  btn.innerText = 'Button';

  btn.className = [
    'storybook-button',
    'storybook-button--medium',
    'storybook-button--primary',
  ].join(' ');

  return btn;
};
